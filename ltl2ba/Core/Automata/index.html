val state_to_string : ?surround:[ `Quotes | `Braces | `Empty ] -> ?empty:string -> state -> string

state_to_string ?surround ?empty state returns the string representation of the state.

~empty is the string to represent an empty state (default "∅").

~surround specify by whitch character the string representation is surrounded (default `Empty).

val formula_map_on_sets_union : StateSet.t FormulaMap.t -> StateSet.t FormulaMap.t -> StateSet.t FormulaMap.t

formula_map_on_sets_union m m' returns the union of each StateSet.t for each corresponding key of m and m'.
